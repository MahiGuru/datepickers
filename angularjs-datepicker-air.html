üìù AngularJS Directive Code

// Air Datepicker Directive for AngularJS app.directive('airDatepicker', ['$timeout', function($timeout) { return { restrict: 'A', require: 'ngModel', scope: { adpOptions: '=', adpOnSelect: '&' }, link: function(scope, element, attrs, ngModel) { var config = angular.copy(scope.adpOptions) || {}; // Sync with ng-model config.onSelect = function({date, formattedDate}) { $timeout(function() { ngModel.$setViewValue(formattedDate); if (scope.adpOnSelect) { scope.adpOnSelect({ date: date, formattedDate: formattedDate }); } }); }; var picker = new AirDatepicker(element[0], config); // Cleanup on destroy scope.$on('$destroy', function() { picker.destroy(); }); } }; }]);
